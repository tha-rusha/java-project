1.public class Throws {
    
     public int divideNum(int m,int n ) throws ArithmeticException
     {
         int div=m/n;
         return div;
     }

    public static void main(String[] args) 
    {
        Throws t=new Throws();
        try
        {
             System.out.println("Answer is : "+t.divideNum(20,0));
        }
       catch (ArithmeticException e)
       {
           System.out.println(" number divide by zero error");
       }
            
        
        
    }
    
}
